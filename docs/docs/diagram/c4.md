```puml
@startuml
!include <C4/C4_Component>

LAYOUT_WITH_LEGEND()

title Диаграмма компонентов: A/B Platform API (Критический поток)

Container_Ext(client_app, "Client Application", "iOS/Android/Web", "Запрашивает флаги, отправляет события")
Container_Ext(admin_ui, "Admin Dashboard", "React/JS", "Настройка тестов, просмотр отчетов")

ContainerDb_Ext(postgres, "PostgreSQL", "Relational DB", "Метаданные: Каталог событий, Схемы, Эксперименты, Метрики")
ContainerDb_Ext(redis, "Redis", "Key-Value Store", "Дедупликация (7д), Кулдауны, Флаги атрибуции")
ContainerDb_Ext(clickhouse, "ClickHouse", "OLAP DB", "События (Accepted/Pending), Решения, Агрегаты для отчетов")

System_Boundary(ab_api, "A/B Platform API Service") {
    
    Component(iap_handler, "IAP Auth Handler", "Middleware", "Извлекает X-User-Id и данные пользователя из заголовков")
    
    ' Блок Принятия решений (Decide)
    Component(decide_controller, "Decide Controller", "REST/JSON API", "Точка входа для батч-запросов флагов")
    Component(decision_engine, "Decision Engine", "Logic", "Детерминированный расчет бакета (Murmurhash3)")
    Component(conflict_resolver, "Conflict Resolver", "Logic", "Разрешение конфликтов (Strict Exclusion / Priority)")
    
    ' Блок Сбора событий (Event)
    Component(event_controller, "Event Ingestion Controller", "REST/JSON API", "Прием пачек событий")
    Component(validator, "Schema Validator", "FastJSONSchema", "Валидация additional_params по скомпилированным схемам")
    Component(deduplicator, "Deduplicator", "Logic", "Проверка уникальности ID события через Redis")
    Component(attr_worker, "Attribution Worker", "Background Process", "Связывает Pending события с родительскими раз в N секунд")

    ' Блок Аналитики (Report/Guardrail)
    Component(report_controller, "Reporting Controller", "REST/JSON API", "Выдача данных для дашбордов и экспорт CSV")
    Component(query_builder, "SQL Query Builder", "Logic", "Генерация SQL на основе каталога метрик (JSONExtract)")

    ' Общий слой данных
    Component(local_cache, "In-Memory Cache", "Sync Map", "Кэш экспериментов, схем и политик (инвалидация по вызову)")
}

' Связи Decide
Rel(client_app, decide_controller, "Запрос флагов (subject_id, attributes, keys)", "HTTPS/JSON")
Rel(decide_controller, iap_handler, "Проверка контекста")
Rel(decide_controller, local_cache, "Получение активных экспериментов")
Rel(decide_controller, redis, "Проверка Cooldown (Sticky)")
Rel(decide_controller, conflict_resolver, "Применение политик разрешения")
Rel(decide_controller, decision_engine, "Хеширование и бакетинг")
Rel(decide_controller, clickhouse, "Логирование Решения (Decision)", "Async Write")

' Связи Event
Rel(client_app, event_controller, "Отправка событий", "HTTPS/JSON")
Rel(event_controller, local_cache, "Получение схемы типа события")
Rel(event_controller, validator, "Валидация структуры")
Rel(event_controller, deduplicator, "Проверка ID", "Redis SETEX")
Rel(event_controller, clickhouse, "Запись событий (Accepted/Pending/Rejected)", "Batch Insert")
Rel_D(attr_worker, clickhouse, "Обновление статусов Pending -> Accepted", "ReplacingMergeTree Logic")

' Связи Report
Rel(admin_ui, report_controller, "Запрос отчета / Guardrails", "HTTPS/JSON")
Rel(report_controller, query_builder, "Запрос структуры метрики")
Rel(query_builder, postgres, "Получение формул и фильтров")
Rel(query_builder, clickhouse, "OLAP агрегация", "SQL (SummingMergeTree)")

' Синхронизация кэша
Rel_U(local_cache, postgres, "Загрузка/Обновление метаданных")

@enduml

```
